# 3.5 旋转数组的最小元素

```php
/**
 * 旋转数组的最小元素
 *  二分法 or 双指针算法
 */

function getMinNum($str, $start, $end){
    echo "{$start}  {$end}  \n";

    if($end - $start == 1){
        return min($str[$start], $str[$end]);
    }
    $middle = intval(($start + $end) / 2);
    if($str[$middle] >= $str[$start]){
        $start = $middle;
    }else{
        $end = $middle;
    }

    return getMinNum($str, $start, $end);
}



function minNum($num){
    echo  getMinNum(strval($num), 0, strlen($num)-1);
}

minNum(45622222333333);
```
