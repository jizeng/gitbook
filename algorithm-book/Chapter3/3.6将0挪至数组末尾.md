# 3.6 将0挪至数组末尾

```php
/**
 *  给定一个数组nums, 写一个函数，将数组中所有的0挪到数组的末尾，而维持其他所有非0元素的相对值。
 *
 *  举例： nums= [0,1,0,2,3]  函数运行后结果为：[1,2,3,0,0]
 *
 */
function arrSwap(&$arr, $i, $start){
    if($i == $start){
        return ;
    }
    $tmp = $arr[$i];
    $arr[$i] = $arr[$start];
    $arr[$start] = $tmp;
    return ;
}

function moveZero($arr){
    //找0的个数
    $numZero = 0;
    foreach ($arr as $v){
        if($v == 0) $numZero++;
    }

    if($numZero==0){
        print_r($arr);
        return ;
    }

    //将非0数据往前挪
    $start = 0;
    $end = count($arr) - $numZero;
    for($i = 0; $i < count($arr); $i++){
        if($start == $end){
            break;
        }
        if($arr[$i] != 0){
            arrSwap($arr, $i, $start);
            $start++;
        }
    }

    //补齐非0 数据
    for($i = $end; $i< count($arr); $i++){
        $arr[$i] = 0;
    }

    print_r($arr);
    return ;
}

moveZero([0,1,0,2,3]);
```
