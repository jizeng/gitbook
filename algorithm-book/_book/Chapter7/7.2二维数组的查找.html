7.2 二维数组的查找

 *  题目：二维数组的查找
 *  题目描述：指定一个二维数组，其每一行从左到右递增排序，从上到到下也是递增排序。给定一个数，判断这个数是否在该二维数组中。

$array = array(
    array(1,4,7,11,15),
    array(2,5,8,12,19),
    array(3,6,9,16,22),
    array(10,13,14,17,24,),
    array(18,21,23,26,30)
);


function searchNum($arr, $num){
    $row = count($arr);
    $col = count($arr[0]);

    $i = $row-1;
    $j = 0;
    //注意php for 里面的逗号 和  && 的区别
    for(; $i >=0  && $j < $col; ){
    //for(; $i >=0  , $j < $col; ){
        echo "{$i}  {$j}  aaa \n";
        if($arr[$i][$j] == $num){
            echo  "{$i}  {$j}";
            return ;
        }elseif ($arr[$i][$j] > $num){
            $i--;
        }else{
            $j++;
        }
    }
    echo  "没有";
    return;
}
